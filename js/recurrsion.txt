1) To calculate factorial of any given element function with arguement with return.
input : 4
Processing : 1*2*3*4
output : 24

#include<stdio.h>
int fact(int);
int main()
{
	int n,x;
	printf("\n Enter a number > ");
	scanf("%d",&x);                 
	n=fact(x);      
	printf("\n Factorial = %d",n);
	return 0;
}
int fact(int a)
{
	int m=1,i;
	for(i=1; i<=a; i++)
		m=m*i;          
	return(m);
}
